Brand Identity

A brand built from zero.

The name, the logo, the color, the font — all original. The logo uses a geometric mark: a three-point pattern (left, middle-right, left) designed to feel cryptic and precise. The primary color is #DE4097 — a hot pink that pops hard against pure black. No softness, no gray. Everything is sharp edges and stark contrast.

Typography: Chakra Petch via Adobe Fonts. A technical, grid-based typeface that feels right at home in a world of blockchain and pixel art. Left-aligned throughout. Nothing centered. Nothing soft.

Button Animation

A canvas pixel-fill effect built from scratch.

The main CTA button fills with pixels on click — a canvas-based animation that tiles 10×10 pixel blocks across the button surface in randomized order, creating a "filling" effect before navigating. Every pixel placed by hand in code.

// Canvas pixel-fill animation
function fillButton(btn, callback) {
  const canvas = document.createElement('canvas');
  const ctx = canvas.getContext('2d');
  const TILE = 10;
  const W = btn.offsetWidth, H = btn.offsetHeight;
  canvas.width = W; canvas.height = H;

  // Build shuffled grid of tile coords
  const cols = Math.ceil(W/TILE), rows = Math.ceil(H/TILE);
  let tiles = [];
  for (let r=0; r<rows; r++)
    for (let c=0; c<cols; c++)
      tiles.push([c*TILE, r*TILE]);

  // Fisher-Yates shuffle
  for (let i=tiles.length-1; i>0; i--) {
    const j = Math.floor(Math.random()*(i+1));
    [tiles[i], tiles[j]] = [tiles[j], tiles[i]];
  }

  ctx.fillStyle = '#DE4097';
  let i = 0;
  const step = () => {
    const batch = Math.ceil(tiles.length / 12);
    for (let b=0; b<batch && i<tiles.length; b++, i++)
      ctx.fillRect(tiles[i][0], tiles[i][1], TILE, TILE);
    if (i < tiles.length) requestAnimationFrame(step);
    else callback();
  };
  requestAnimationFrame(step);
}
